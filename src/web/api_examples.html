Get All Active Agents

GET /v2/agents
JavaScript (ES6) 
// Get all active agents
const apiKey = 'your-api-key-here';
const baseURL = 'https://api.swarm.intelligence/v2';

async function getAllAgents() {
    try {
        const response = await fetch(`${baseURL}/agents`, {
            method: 'GET',
            headers: {
                'X-API-Key': apiKey,
                'Accept': 'application/json'
            }
        });

        if (!response.ok) {
            throw new Error(`HTTP error! status: ${response.status}`);
        }

        const result = await response.json();
        console.log('Active agents:', result.data);
        return result.data;
    } catch (error) {
        console.error('Error fetching agents:', error);
        throw error;
    }
}

// Usage
getAllAgents()
    .then(agents => {
        agents.forEach(agent => {
            console.log(`${agent.agent_name} (${agent.agent_id}): ${agent.status}`);
        });
    })
    .catch(error => console.error('Failed to get agents:', error));
Python (requests) 
import requests
from typing import List, Dict, Any

def get_all_agents(api_key: str, base_url: str = "https://api.swarm.intelligence/v2") -> List[Dict[str, Any]]:
    """
    Retrieve all active agents from the Swarm API.

    Args:
        api_key: Your API key for authentication
        base_url: Base URL for the API

    Returns:
        List of agent dictionaries

    Raises:
        requests.RequestException: If the API request fails
    """
    url = f"{base_url}/agents"
    headers = {
        "X-API-Key": api_key,
        "Accept": "application/json"
    }

    try:
        response = requests.get(url, headers=headers)
        response.raise_for_status()

        result = response.json()
        return result.get("data", [])

    except requests.RequestException as e:
        print(f"Error fetching agents: {e}")
        raise

# Usage example
if __name__ == "__main__":
    API_KEY = "your-api-key-here"

    try:
        agents = get_all_agents(API_KEY)
        for agent in agents:
            status = agent.get("status", "UNKNOWN")
            name = agent.get("agent_name", "Unnamed Agent")
            agent_id = agent.get("agent_id", "unknown")
            print(f"Agent {agent_id}: {name} - Status: {status}")

    except Exception as e:
        print(f"Failed to retrieve agents: {e}")

JavaScript (ES6) 
// Send a message to Agent-7
const apiKey = 'your-api-key-here';
const baseURL = 'https://api.swarm.intelligence/v2';

async function sendMessage(toAgent, content, priority = 'NORMAL', tags = []) {
    try {
        const response = await fetch(`${baseURL}/messages`, {
            method: 'POST',
            headers: {
                'Authorization': `Bearer ${apiKey}`,
                'Content-Type': 'application/json',
                'Accept': 'application/json'
            },
            body: JSON.stringify({
                to_agent: toAgent,
                content: content,
                priority: priority,
                tags: tags
            })
        });

        if (!response.ok) {
            throw new Error(`HTTP error! status: ${response.status}`);
        }

        const result = await response.json();
        console.log('Message sent successfully:', result);
        return result;
    } catch (error) {
        console.error('Error sending message:', error);
        throw error;
    }
}

// Usage examples
async function sendExamples() {
    try {
        // Send a task assignment
        await sendMessage(
            'Agent-7',
            'Please review the latest API documentation updates',
            'NORMAL',
            ['documentation', 'review']

Python (requests) 
import requests
from typing import List, Optional
from enum import Enum

class MessagePriority(Enum):
    LOW = "LOW"
    NORMAL = "NORMAL"
    HIGH = "HIGH"
    URGENT = "URGENT"

def send_message(
    to_agent: str,
    content: str,
    api_key: str,
    priority: MessagePriority = MessagePriority.NORMAL,
    tags: Optional[List[str]] = None,
    base_url: str = "https://api.swarm.intelligence/v2"
) -> dict:
    """
    Send a message to another agent via the Swarm API.

    Args:
        to_agent: ID of the recipient agent
        content: Message content
        api_key: Your API key for authentication
        priority: Message priority level
        tags: Optional list of message tags
        base_url: Base URL for the API

    Returns:
        API response containing message details

    Raises:
        requests.RequestException: If the API request fails
    """
    url = f"{base_url}/messages"
    headers = {
        "Authorization": f"Bearer {api_key}",
        "Content-Type": "application/json",
        "Accept": "application/json"
    }

    payload = {
        "to_agent": to_agent,
        "content": content,
        "priority": priority.value,
        "tags": tags or []
    }

    try:
        response = requests.post(url, json=payload, headers=headers)
        response.raise_for_status()

        result = response.json()
        return result

    except requests.RequestException as e:
        print(f"Error sending message: {e}")
        raise

# Usage examples
if __name__ == "__main__":
    API_KEY = "your-api-key-here"

    try:
        # Send a coordination message
        result = send_message(
            to_agent="Agent-7",
            content="API documentation enhancement task completed successfully",
            api_key=API_KEY,
            priority=MessagePriority.NORMAL,
            tags=["api", "documentation", "completion"]
        )
        print(f"Message sent! ID: {result['data']['message_id']}")

        # Send a broadcast message (would require different endpoint)
        print("Broadcast messages would use /messages/broadcast endpoint")

    except Exception as e:
        print(f"Failed to send message: {e}")

// Perform semantic similarity search
const apiKey = 'your-api-key-here';
const baseURL = 'https://api.swarm.intelligence/v2';

async function vectorSearch(query, limit = 10, threshold = 0.7) {
    try {
        const response = await fetch(`${baseURL}/vector/search`, {
            method: 'POST',
            headers: {
                'X-API-Key': apiKey,
                'Content-Type': 'application/json',
                'Accept': 'application/json'
            },
            body: JSON.stringify({
                query: query,
                limit: limit,
                threshold: threshold
            })
        });

        if (!response.ok) {
            throw new Error(`HTTP error! status: ${response.status}`);
        }

        const result = await response.json();
        console.log('Search results:', result.data);
        return result.data;
    } catch (error) {
        console.error('Error performing vector search:', error);
        throw error;
    }
}

async function searchExamples() {
    try {
        // Search for agent coordination patterns
        console.log('Searching for agent coordination patterns...');
        const coordinationResults = await vectorSearch(
            'agent coordination and task management patterns',
            5,
            0.8
        );

        coordinationResults.forEach((result, index) => {
            console.log(`${index + 1}. ${result.content.substring(0, 100)}...`);
            console.log(`   Similarity: ${(result.similarity_score * 100).toFixed(1)}%`);
        });

        // Search for API documentation
        console.log('\nSearching for API documentation...');
        const apiResults = await vectorSearch(
            'REST API endpoints and integration guides',
            3,
            0.75
        );

        apiResults.forEach((result, index) => {
            console.log(`${index + 1}. ${result.content.substring(0, 80)}...`);
        });

    } catch (error) {
        console.error('Search failed:', error);
    }
}

// Execute search examples
searchExamples();

import requests
from typing import List, Dict, Any, Optional

def vector_similarity_search(
    query: str,
    api_key: str,
    limit: int = 10,
    threshold: float = 0.7,
    filters: Optional[Dict[str, Any]] = None,
    base_url: str = "https://api.swarm.intelligence/v2"
) -> List[Dict[str, Any]]:
    """
    Perform semantic similarity search using the Swarm vector database.

    Args:
        query: Search query string
        api_key: Your API key for authentication
        limit: Maximum number of results to return
        threshold: Similarity threshold (0-1)
        filters: Optional metadata filters
        base_url: Base URL for the API

    Returns:
        List of search results with similarity scores

    Raises:
        requests.RequestException: If the API request fails
    """
    url = f"{base_url}/vector/search"
    headers = {
        "X-API-Key": api_key,
        "Content-Type": "application/json",
        "Accept": "application/json"
    }

    payload = {
        "query": query,
        "limit": limit,
        "threshold": threshold
    }

    if filters:
        payload["filters"] = filters

    try:
        response = requests.post(url, json=payload, headers=headers)
        response.raise_for_status()

        result = response.json()
        return result.get("data", [])

    except requests.RequestException as e:
        print(f"Error performing vector search: {e}")
        raise

def add_vector_document(
    content: str,
    api_key: str,
    metadata: Optional[Dict[str, Any]] = None,
    collection: str = "default",
    base_url: str = "https://api.swarm.intelligence/v2"
) -> Dict[str, Any]:
    """
    Add a document to the vector database for future similarity searches.

    Args:
        content: Document content to index
        api_key: Your API key for authentication
        metadata: Optional metadata for filtering and categorization
        collection: Collection name for organization
        base_url: Base URL for the API

    Returns:
        Document creation confirmation

    Raises:
        requests.RequestException: If the API request fails
    """
    url = f"{base_url}/vector/documents"
    headers = {
        "Authorization": f"Bearer {api_key}",
        "Content-Type": "application/json",
        "Accept": "application/json"
    }

    payload = {
        "content": content,
        "metadata": metadata or {},
        "collection": collection
    }

    try:
        response = requests.post(url, json=payload, headers=headers)
        response.raise_for_status()

        result = response.json()
        return result

    except requests.RequestException as e:
        print(f"Error adding document: {e}")
        raise
